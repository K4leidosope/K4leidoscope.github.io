class INT {
public:
  int a[M];
  INT() { memset(a, 0, sizeof(a)); }
  void clear() { memset(a, 0, sizeof(a)); }
  INT(int x) {
    clear();
    while (x) {
      a[++a[0]] = x % 10;
      x /= 10;
    }
    while (a[a[0]] == 0 && a[0])
      a[0]--;
  }
  INT &operator=(int x) {
    clear();
    while (x) {
      a[++a[0]] = x % 10;
      x /= 10;
    }
    while (a[a[0]] == 0 && a[0])
      a[0]--;
    return *this;
  }

  short cmp(const INT &x) {
    if (a[0] > x.a[0])
      return 1;
    if (a[0] < x.a[0])
      return -1;
    for (register int i = a[0]; i >= 1; i--) {
      if (a[i] > x.a[i])
        return 1;
      if (a[i] < x.a[i])
        return -1;
    }
    return 0;
  }
  bool operator>(const INT &x) { return cmp(x) == 1; }
  bool operator==(const INT &x) { return cmp(x) == 0; }
  bool operator<(const INT &x) { return cmp(x) == -1; }
  bool operator>=(const INT &x) { return !(*this < x); }
  bool operator<=(const INT &x) { return !(*this > x); }

  INT operator-(const INT &x) {
    INT a = *this, c;
    c.a[0] = a.a[0] > x.a[0] ? a.a[0] : x.a[0];
    for (register int i = 1; i <= c.a[0]; i++) {
      c.a[i] += a.a[i] - x.a[i];
      if (c.a[i] < 0) {
        c.a[i] += 10;
        a.a[i + 1]--;
      }
    }
    while (c.a[c.a[0]] == 0 && c.a[0])
      c.a[0]--;
    return c;
  }
  INT operator*(const INT &x) {
    INT c;
    for (register int i = 1; i <= a[0]; i++) {
      for (register int j = 1; j <= x.a[0]; j++) {
        c.a[i + j - 1] += a[i] * x.a[j];
      }
    }
    c.a[0] = a[0] + x.a[0];
    for (register int i = 1; i <= c.a[0]; i++) {
      if (c.a[i] >= 10) {
        c.a[i + 1] += c.a[i] / 10;
        c.a[i] %= 10;
      }
    }
    while (c.a[c.a[0]] == 0 && c.a[0] > 0)
      c.a[0]--;
    return c;
  }
  INT operator/(const int &x) {
    INT c;
    int t = 0, s = 0;
    bool flag = 1;
    for (register int i = a[0]; i >= 1; i--) {
      t = s * 10 + a[i];
      if (t / x > 0 || t == 0) {
        c.a[++c.a[0]] = t / x;
        s = t % x;
        flag = 0;
      } else {
        s = t;
        if (!flag)
          c.a[++c.a[0]] = 0;
      }
    }
    reverse(c.a + 1, c.a + c.a[0] + 1);
    return c;
  }
};
